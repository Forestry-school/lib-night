function doGet(e) { const action = e.parameter.action; const ss = SpreadsheetApp.getActiveSpreadsheet(); // 1. 取得學生名單，並同步比對「今日已登記」狀態 if (action === 'getStudents') { const sheet = ss.getSheetByName("學生名單"); const data = sheet.getDataRange().getDisplayValues(); // 讀取今日已登記名單 const logSheet = ss.getSheetByName("晚接紀錄"); const logData = logSheet.getDataRange().getDisplayValues(); const now = new Date(); const dateStr = Utilities.formatDate(now, "Asia/Taipei", "yyyy/MM/dd"); // 建立一個集合，記錄今天已經登記的「班級-座號」 let todaySet = new Set(); for(let i = 1; i < logData.length; i++) { if(logData[i][0] === dateStr) { let c = String(logData[i][2]).trim(); let s = String(logData[i][3]).trim().replace(/'/g, '').padStart(2, '0'); todaySet.add(c + '-' + s); } } let students = []; for (let i = 1; i < data.length; i++) { if(data[i][0] !== "") { let c = String(data[i][0]).trim(); let s = String(data[i][1]).trim().padStart(2, '0'); let n = String(data[i][2]).trim(); // 判斷該生今天是否已經在紀錄中 let isReg = todaySet.has(c + '-' + s); students.push({ className: c, seat: s, name: n, isRegistered: isReg }); } } return ContentService.createTextOutput(JSON.stringify(students)).setMimeType(ContentService.MimeType.JSON); } // 2. 新增或更新晚接紀錄 if (action === 'addLog') { const sheet = ss.getSheetByName("晚接紀錄"); const className = String(e.parameter.className).trim(); const name = String(e.parameter.name).trim(); const incomingSeat = String(e.parameter.seat).trim().padStart(2, '0'); const now = new Date(); const dateStr = Utilities.formatDate(now, "Asia/Taipei", "yyyy/MM/dd"); const timeStr = Utilities.formatDate(now, "Asia/Taipei", "HH:mm:ss"); const data = sheet.getDataRange().getDisplayValues(); let isDuplicate = false; let targetRow = -1; for (let i = 1; i < data.length; i++) { let rowDate = String(data[i][0]).trim(); let rowClass = String(data[i][2]).trim(); let rowSeat = String(data[i][3]).trim().replace(/'/g, '').padStart(2, '0'); if (rowDate === dateStr && rowClass === className && rowSeat === incomingSeat) { isDuplicate = true; targetRow = i + 1; break; } } // 寫入時加上單引號，避免試算表自動將 02 轉成 2 const safeSeatToSave = "'" + incomingSeat; if (isDuplicate) { sheet.getRange(targetRow, 2).setValue(timeStr); sheet.getRange(targetRow, 4).setValue(safeSeatToSave); return ContentService.createTextOutput(JSON.stringify({status: "success", name: name, message: "updated"})).setMimeType(ContentService.MimeType.JSON); } else { sheet.appendRow([dateStr, timeStr, className, safeSeatToSave, name]); return ContentService.createTextOutput(JSON.stringify({status: "success", name: name, message: "added"})).setMimeType(ContentService.MimeType.JSON); } } // 3. 刪除今天的誤按紀錄 if (action === 'removeLog') { const sheet = ss.getSheetByName("晚接紀錄"); const className = String(e.parameter.className).trim(); const incomingSeat = String(e.parameter.seat).trim().padStart(2, '0'); const now = new Date(); const dateStr = Utilities.formatDate(now, "Asia/Taipei", "yyyy/MM/dd"); const data = sheet.getDataRange().getDisplayValues(); // 從最後一筆往前找，找到今天該名學生的紀錄並刪除 for (let i = data.length - 1; i >= 1; i--) { let rowDate = String(data[i][0]).trim(); let rowClass = String(data[i][2]).trim(); let rowSeat = String(data[i][3]).trim().replace(/'/g, '').padStart(2, '0'); if (rowDate === dateStr && rowClass === className && rowSeat === incomingSeat) { sheet.deleteRow(i + 1); return ContentService.createTextOutput(JSON.stringify({status: "success", message: "deleted"})).setMimeType(ContentService.MimeType.JSON); } } return ContentService.createTextOutput(JSON.stringify({status: "not_found"})).setMimeType(ContentService.MimeType.JSON); } } /** * 自動產生晚接統計表 (供試算表公式使用) */ function GENERATE_REPORT(data) { if (!data || data.length === 0) return [["無資料"]]; let summary = {}; for (let i = 0; i < data.length; i++) { let date = data[i][0]; let className = data[i][2]; let seat = data[i][3]; let name = data[i][4]; // 跳過空白或標題列 if (!className || !name || className === "班級") continue; // 建立唯一鍵值：班級_座號_姓名 let key = className + "_" + String(seat).padStart(2, '0') + "_" + name; if (!summary[key]) { summary[key] = { className: className, seat: String(seat).padStart(2, '0'), name: name, count: 0, dates: [] }; } summary[key].count += 1; // 格式化日期，只保留月份與日期 (MM/dd) 以節省報表空間 let dateStr = ""; if (date instanceof Date) { dateStr = Utilities.formatDate(date, "Asia/Taipei", "MM/dd"); } else { dateStr = String(date).split(" ")[0].substring(5); // 例如將 "2026/05/19" 切割為 "05/19" } if (!summary[key].dates.includes(dateStr) && dateStr) { summary[key].dates.push(dateStr); } } let result = [["班級", "座號", "姓名", "次數", "晚接日期記錄"]]; let sortedKeys = Object.keys(summary).sort(); for (let i = 0; i < sortedKeys.length; i++) { let k = sortedKeys[i]; let s = summary[k]; result.push([ s.className, "'" + s.seat, s.name, s.count, s.dates.join(", ") ]); } return result; }
